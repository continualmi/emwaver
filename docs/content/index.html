Wavelets: One Script, Every Surface

Wavelets are powered by the EMWaver DSL so the same script renders consistent UI in the Wavelets fragment, iOS, Android, and desktop. Author once, sync through Git, and automate hardware interactions everywhere.

EMWaver DSL
ir_send_saved_signal.emw
const DEFAULT_FREQ_HZ = 38000;
const DEFAULT_DUTY_PERCENT = 50;

let selectedSignal = "";
let repetitionsText = "1";
let intervalText = "60";
let status = "Ready";

function sendSignal() {
    status = "Sending...";
    render();
    Utils.delay(10);

    const data = SamplerSignals.readSignal(selectedSignal);
    const cmd = "transmit start --pin=4 --pwm --freq=" + DEFAULT_FREQ_HZ +
        " --duty=" + DEFAULT_DUTY_PERCENT + "\n";

    BLEService.loadBuffer(data);
    BLEService.write(bytesFromString(cmd));
    BLEService.transmitBuffer();

    status = "Sent";
    render();
}

function render() {
    UI.render(UI.column({
        spacing: 12,
        children: [
            UI.text({ text: "IR: Send Saved Signal", font: "title2" }),
            UI.picker({ ... }),
            UI.textField({ label: "Repetitions", value: repetitionsText }),
            UI.button({ label: "Send", onTap: sendSignal })
        ]
    }));
}

render();
